#INCLUDE <STDIO.H>
// THE CTYPE.H HEADER FILE OF THE C STANDARD LIBRARY DECLARES SEVERAL FUNCTIONS THAT ARE USEFUL FOR TESTING AND MAPPING CHARACTERS.
#INCLUDE <CTYPE.H>

INT MAIN()
{
    INT C;
    WHILE ((C = GETCHAR()) != EOF)
        PUTCHAR(TOUPPER(C));

    RETURN 0;
}
